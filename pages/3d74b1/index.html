<script>
const { gsap } = require('gsap')
export default {
  name: 'GSAPHeartShapeCardTransform',
  mounted() {
    this.runAnimation()
  },
  methods: {
    runAnimation() {
      gsap.to('.heart_area .heart_img', {
        scale: 0.3, //! 缩放
        y: 20,
        duration: 2, //! 动画持续时间
        repeat: -1, //? 重复次数
        yoyo: true, //todo 进行往返放大缩小
        delay: 2, //* 动画延迟开启时间
        // stagger: 0.1 //* 简单配置
        // 对象配置
        stagger: {
          amount: 2,
          grid: 'auto',
          from: 'center'
        }
      })
    }
  }
}
</script>

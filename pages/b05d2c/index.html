  .demo {
    font-size: 20px;
    width: 500px;
    line-height: 40px;
  }
  .demo span {
    background: linear-gradient(to right, red, green, blue) no-repeat right bottom;
    background-size: 0 3px;
    transition: background-size 1300ms;
  }
  .demo:hover span {
    background-position-x: left;
    background-size: 100% 3px;
  }
</style>
